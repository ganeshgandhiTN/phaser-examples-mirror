}

    cropRect = new Phaser.Rectangle(0, 100, 320, 50);
    
    irect = Phaser.Rectangle.intersection(cropRect, fakeImage);

}

}

    game.debug.geom(fakeImage, 'rgba(0,255,0,0.3)');

    game.debug.geom(irect, 'rgba(255,255,0,1)');

}
